:first-child / :last-child / :nth-child() / :nth-last-child() / :first-of-type / last-of-type / nth-of-type() / nth-last-of-type()

SYNTAX :    element:first-child                                        // selects the first targeted element of a group (if the element type does not match it is NOT selected)  
            element:last-child                                         //  selects the last targeted element of a group (if the element type does not match it is NOT selected)  
            element:nth-child(nr|even|odd|nrn|cycle_nrn + nr)          //  selects element(s) in the group or in every cycle (if element(s) type do not match they are NOT selected)
            element:nth-last-child(nr|even|odd|nrn|cycle_nrn + nr)     //  selects element(s) in the group or in every cycle (counted form the end-to-start) (if element(s) type do not match they are NOT selected)

            element:first-of-type                                      //  selects the first element with the corresponding type of a group of elements 
            element:last-of-type                                       //  selects the last element with the corresponding type of a group of elements 
            element:nth-of-type(nr|even|odd|nrn|cycle_nrn + nr)        //  selects the element(s) in the group or in the cycle with the corresponding type 
            element:nth-last-of-type(nr|even|odd|nrn|cycle_nrn + nr)   //  selects the element(s) in the group or in the cycle with the corresponding type (counted form the end-to-start)

                - nr|even|odd|(nr|even|odd|nrn|cycle_nrn + nr) : nr                 - targeted element in the sequence
                                               : even               - every even element
                                               : odd                - every odd element
                                               : nrn                - every nr element 
                                               : cycle_nrn + nr     - cycle_nrn (cycle size) | nr (targeted element in each cycle) 
    
    <p> paragraph_1 </p>
    <span> span_1 </span> 
    <div> div_1 </div>
    <div> div_2 </div>
    <p> paragraph_2 </p>
    <p> paragraph_3 </p>
    <span> span_2 </span>
    <div>> div_3 </div>
    <span> span_3 </span>
        
-- CSS -- 
// :first child / :last child / :first-of-type / :last-of-type -----------------
    p:first-child { color: red; }                                      // "paragraph_1" selected / 
    p:last-child { color: red; }                                       // nothing selected because the first element is NOT a "p" element 
    div:first-of-type { color: yellow; }                               // "div_1"  selected  
    span:last-of-type {color: yellow; }                                // "span_3" selected
    
// :nth-child() / nth-of-type() / nth-last-child() / nth-last-of-type() --------
    *:nth-child(odd) { color: red; }                                   // "paragraph_1", "div_1", "paragraph_2", "span_2", "span_3" selected (every odd element)
    span:nth-of-type(2) { color: red; }                                // "span_2" selected
    *:nth-last-child(3n+3) { color: red; }                             // "span_2", "div_2", "paragraph_1" selected // in every 3 elements(cycle size 3) the 3rd one (counted from the end)   
    p:nth-last-of-type(2) { color:red; }                               // "paragraph_2" selected

:not() / :empty / :only-child / :only-of-type 

SYNTAX :    element:not(selector)                                      //  selects every element that do NOT match the specified selector
            element:empty                                              //  select elements that has no child nodes
            
            element:only-child                                         //  selects the element that is the only child of its parent
            element:only-of-type                                       //  selects the element that has the only type of the group
    
    <p>paragraph_1</p>
    <span> span_1 <span>child span element</span> </span>
    <p>paragraph_2</p>
    <div>div_1</div>
    <p></p>
    
-- CSS --
// only-of-type -------------------------------
    :only-of-type { color: red; }                                      // "<span> span_1 <span>child span element</span> </span>", "<span>child span element</span>" selected (because they are only children with no sibling element with the same type)   
    p:only-of-type { color: red; }                                     // nothing selected because there are multiple "p" elements  
    
// only-child ---------------------------------
    div:only-child { color: red; }                                     // nothing selected because there are multiple elements in the groupe (the "div" element is not the only child) 
    :only-child { color: red; }                                        // "<span>child span element</span>" selected (element with no siblings)
    
// not() --------------------------------------
    :not(div, span) { color: red; }                                    // "<p>paragraph_1</p>", "<p>paragraph_2</p>", "<div>div_1</div>", "<p></p>" selected (every other element that the targeted ones)
    :not(div, span) { color: red; }                                    <- this is not working (":not()" pseudo-class accepts only 1 selector!) 
    
// empty -------------------------------------- 
    :empty {border: solid 2px red;}                                    // "<p></p>" selected

SYNTAX :    element:default                                            //  selects the default submit button in the from (can be <button>, <input type="submit"> or <input type="image>) 
                                                                          selects the <option> element that has the "selected" attribute
                                                                          selects the <input type="checkbox> or <input type="radio"> elements that has the "checked" attribute

            element:disabled                                           //  selects disabled elements (element cannot be clicked, typed into, ect..) (mostly form elements)

            element:required                                           //  selects form elements which have the "required" attribute
