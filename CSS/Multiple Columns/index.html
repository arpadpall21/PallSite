Description and Demonstration : 

SYNTAX :    column-count: nr                                    //  specifies the number of columns the element should divided into (maximum number of columns)  
                          auto ((default) determined by other properties)
            
            column-gap: cssLength                               //  specifies the gap size between columns 
                        normal ((default) 1em suggested by W3C)
                        
            column-width: cssLength                             //  specifies the minimum column width  
                          auto ((default) determined by the browser)
                        
            columns: auto (default)                             //  shorthand property for the "column-width" and "column-count" properties (sets the minimal column width and the maximal column numbers)     
                     column-width column-count
                        
            column-span: none (default)                         //  specifies on the nested element how many columns should span across 
                         all (span accross all columns)
                        
            column-fill: balance ((default) distributes the content among the columns evenly (columns can be shorter than the element's height but NOT taller))  //  how to distribute the content among columns   
                         auto (fills each columns up to the element's height until it runs out of content)
                        
            column-rule-width: medium (default)                 //  width of the rule between columns 
                               thin
                               thick
                               cssLength
                        
            column-rule-color: cssColor                         //  color of the rule between columns
                        
            column-rule-style: none (default)                   //  style of the rule between columns 
                               hidden
                               dotted
                               dashed
                               solid
                               double
                               groove
                               ridge
                               inset
                               outset
            
            column-rule: column-rule-width column-rule-style column-rule-color     //  shorthand property for the "column-rule-width", "column-rule-style" and "column-rule-color" properties 
    
    #myElement {
// column-width / column-count  --------------------------------- 
        column-width: 200px;                                    // minimal column width (can be wider)
        column-count: 4;                                        // no more than 4 columns (can be less)
// columns ------------------------------------------------------ 
        columns: 200px 4;                                       // does the same as the two properties above 
        
// column-rule-width / column-rule-style / column-rule-color ----  
        column-rule-width: 5px;                                 // width of the rule between the columns 
        column-rule-style: double;                              // style of the rule between the columns
        column-rule-color: red;                                 // color of the rule between the columns
// column-rule -------------------------------------------------- 
        column-rule: 5px double red;                            // does the same as the three properties above 
    
// column-gap --------------------------------------------------- 
        column-gap: 5px;                                        // gap between columns 
    
// column-fill -------------------------------------------------- 
        column-fill: balance;                                   // content distributed evenly (can be shorter than the element's height in order to be distributed evenly)  
                   : auto;                                      // content fills the element's height until it runs out 
    }
    
// column-span -------------------------------------------------- 
    #myElement > p {
        column-span: all;                                       // children element crosses all the columns 
    }
