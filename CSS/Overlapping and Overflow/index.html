Overflow overflow / overflow-x / overflow-y 

SYNTAX :    overflow: visible           // (default) content is visible outside of the container   // specifies what to do when the content is too big to fit in its container   
                    : hidden            // content is not visible outside the container 
                    : scroll            // scroll bar added, content is not visible outside the container 
                    : auto              // if content overflows adds scroll bar 
            
            overflow-x: visible         // (default) content is visible outside of the container   // specifies what to do horizontally when the content is too big to fit in its container   
                    : hidden            // content is not visible outside the container 
                    : scroll            // scroll bar added, content is not visible outside the container 
                    : auto              // if content overflows adds scroll bar 
        
            overflow-y: visible         // (default) content is visible outside of the container   // specifies what to do vertically when the content is too big to fit in its container   
                    : hidden            // content is not visible outside the container 
                    : scroll            // scroll bar added, content is not visible outside the container 
                    : auto              // if content overflows adds scroll bar 
    
    .myElement { 
        overflow: auto;                 // if the content is too big to fit in the container the browser automatically adds a scroll bar to the element
    }

Overlap z-index 

SYNTAX :    z-index: auto               // (default) equal to its parents        // defines the stacking order when the element is overlapping, (highest value is closer to the viewer) (does not work if position:static)   
                   : nr                 // stack order, negative number is allowed
    
-- HTML -- 
    <p id="demo"> demo element 1 </p>
    <p> demo element 2 </p>                                                      // by default the latest element overlaps the former ones 
        
-- CSS -- 
    #demo {
        position: absolute;             // -! position other than "static" must be declared in order to the "z-index" to work (this element floats over the layout)
        z-index: 1;                     // visible over the "<p>demo element 2</p>" element (closer to the viewer)
    }
