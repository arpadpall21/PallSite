Demo 2

CODE

    <div class="txt">
        <form>
            <textarea name="text">
            </textarea>
        </form>
    </div>
    
-- CODE (CSS) --
    .txt{                                           .txt textarea:focus{
        width: 60%;                                     border-color: white;
        margin: auto;                               }
    }
    .txt textarea{
        width: 100%;
        background-color: lightblue;
        outline: none;
        height: 100px;
        resize: none;
        border: black solid 2px;
        transition: border 750ms;
    }
