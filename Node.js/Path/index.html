Node.js path module

Remember This :

- POSIX uses /(slash) and windows uses both /(slash) and \(backslash) as path segment separator

Description and Demonstration :

- the path module provides utilities for working with file and directory paths

- Windows and POSIX uses different path forms, Node.js detects the current platform and uses the current platform specific path methods

- we can explicitly use windows specific methods through the win32 property and POSIX specific methods through the posix property

Methods and Properties

The win32 / posix / sep / delimiter properties

DEMO 
    const path = require('path');
    
// win32 / posix ----------------------------------------------------------------
    path.isAbsolute('d:\\Safe\\CODE +\\My Site\\');                              // -> true         // detected platform = windows 
    path.win32.isAbsolute('d:\\Safe\\CODE +\\My Site\\');                        // -> true         // this is an absolute path on windows 
    path.posix.isAbsolute('d:\\Safe\\CODE +\\My Site\\');                        // -> false        // not an absolute path on posix (because the path is windows specific)   
    
// sep / delimiter --------------------------------------------------------------
    path.win32.sep;                                                              // -> '\'          // windows path segment separator 
    path.posix.sep;                                                              // -> '/'          // posix path segment separator   
    
    path.win32.delimiter                                                         // -> ';'          // windows path delimiter 
    path.posix.delimiter                                                         // -> ':'          // posix path delimiter

The dirname() / basename() / extname() methods

DEMO 
    const path = require('path');
    
// dirname() -------------------------------------------------------------------- 
    path.win32.dirname('C:\\Program Files\\node\\node.js');                      // -> 'C:\Program Files\node'          // directy path of the passed file path returned    
    path.win32.dirname('C:\\Program Files\\node\\');                             // -> 'C:\Program Files'               // -! not a file path = previous directory path returned 
    
    path.posix.dirname('/dev/node/node.js');                                     // -> '/dev/node'
    path.posix.dirname('/dev/node/npm/color_module/bin');                        // -> '/dev/node/npm/color_module'     // -! not a file path = previous directory path returned 
    
// basename() -------------------------------------------------------------------
    path.win32.basename('C:\\Program Files\\nodeFolder\\node.js');               // -> 'node.js'                        // returns the last portion of the path  
    path.win32.basename('C:\\Program Files\\nodeFolder');                        // -> 'nodeFolder'
    path.win32.basename('C:\\Program Files\\nodeFolder\\node.js', '.js');        // -> 'node'                           // removes the extension of the returned result 
    path.win32.basename('C:\\Program Files\\nodeFolder', 'er');                  // -> 'nodeFold'                       // the 2nd argument is basically removed from the end of the returned result   
    
    path.posix.basename('/dev/nodeFolder/node.js');                              // -> 'node.js'
    path.posix.basename('/dev/nodeFolder');                                      // -> 'nodeFolder'
    path.posix.basename('/dev/nodeFolder/node.js', '.js');                       // -> 'node'
    path.posix.basename('/dev/nodeFolder', 'deFolder');                          // -> 'no'                             // the 2nd argument is basically removed from the end of the returned result   
    
// extname() --------------------------------------------------------------------
    path.win32.extname('C:/Program Files/nodeFolder/node.js');                   // -> '.js'                            // returns the file extension 
    path.win32.extname('C:/Program.Files/node.Folder');                          // -> '.Folder'                        // basically returns the string portion after the '.'(period) character in the last portion of the path    
    
    path.posix.extname('dev/nodeFolder/node.js');                                // -> '.js'
    path.posix.extname('dev/Program.Files');                                     // -> '.Files'

The parse() / format() methods

DEMO 
    const path = require('path');
    
// parse() ----------------------------------------------------------------------
    path.win32.parse('D:\\serv\\html\\index.html')                               // -> {root:'D:\\', dir:'D:\\serv\\html', base:'index.html', ext:'.html', name:'index'}
    path.posix.parse('/srv/html/index.html');                                    // -> {root:'/', dir:'/srv/html', base:'index.html', ext:'.html', name:'index'}
    
// format() ---------------------------------------------------------------------
    path.win32.format({dir:'D:\\serv\\html',                                     // -> 'D:\serv\html\index.html'
                       root:'D:\\', 
                       base:'index.html', 
                       name:'index', 
                       ext:'.html'});
    
    path.posix.format({dir:'/serv/html',                                         // -> '/serv/html/index.html'      // -! the 'dir' property overrides the 'root' property 
                       root:'\\',                                                                                   // this is ignored 
                       base:'index.html',                                                                           // -! the 'base' property overrides the 'name' and 'ext' properties   
                       name:'something other',                                                                      // this is ignored
                       ext:'.css'})                                                                                 // this is ignored

The normalize() / join() / resolve() methods

- join() simply joins path segments together VS resolve() resolves path segments into an absolute path

DEMO 
    const path = require('path');
    
// normalize() ------------------------------------------------------------------
    path.win32.normalize('C:\\Program Files\\Call of Duty\\..');                 // -> 'C:\Program Files'                           // normalized to the parent directory 
    path.win32.normalize('C:\\Program Files\\Call of Duty\\.');                  // -> 'C:\Program Files\Call of Duty'              // normalized to the current directory  
    path.win32.normalize('C:////Program Files\\\\Call of Duty///');              // -> 'C:\Program Files\Call of Duty\'             // multiple path segment separators are reduces (normalized) to one platform specific and trailing allowed   
    
    path.posix.normalize('dev/appliatons/Call of Duty/..');                      // -> 'dev/appliatons'                             // normalized to the parent directory 
    path.posix.normalize('dev/appliatons/Call of Duty/.');                       // -> 'dev/appliatons/Call of Duty'                // normalized to the current directory  
    path.posix.normalize('dev///applications/////Call of Duty//');               // -> 'dev/applications/Call of Duty/'             // multiple path segment separators are reduces (normalized) to one platform specific and trailing allowed   
    
// join() -----------------------------------------------------------------------
    path.win32.join('/D:', '//Program Files\\\\', 'Call of Duty', 'run.exe//');  // -> '\D:\Program Files\Call of Duty\run.exe\'    // joins path segments together then uses the 'normalize()' method on the final result 
    
    path.posix.join('/dev', '//Program Files////', 'Call of Duty', 'run.exe//'); // -> '/dev/Program Files/Call of Duty/run.exe/'   // joins path segments together then uses the 'normalize()' method on the final result 
    
// resolve() --------------------------------------------------------------------
    path.win32.resolve('\\', 'Program File', 'Call of Duty', 'run.exe');         // -> 'D:\Program File\Call of Duty\run.exe'       // resolves to an absolute path then uses the 'normalize' method on the final result 
    path.win32.resolve('C:', 'Program File\\', 'Call of Duty', 'run.exe//');     // -> 'C:\Program File\Call of Duty\run.exe'       // no segment separator trailing (obviously this is an absolute path)   
    path.win32.resolve();                                                        // -> 'current absolute path'                      // if no arguments are passed the current absolute path will be returned  
    
    path.posix.resolve('/srv', '/dev', 'app', 'Call of Duty', 'run.exe');        // -> '/dev/app/Call of Duty/run.exe'              // -! on posix the last path segment starting with the '/'(backslash) character will be the root 
    path.posix.resolve('/srv', '/dev', 'app', 'Call of Duty', 'run.exe////');    // -> '/dev/app/Call of Duty/run.exe'              // no segment separator trailing (obviously this is an absolute path)   
    path.posix.resolve();                                                        // -> 'current absolute path'                      // if no arguments are passed the current absolute path will be returned

The relative() / isAbsolute() / toNamespacedPath() methods

DEMO 
    const path = require('path');
    
// relative() ------------------------------------------------------------------
    path.win32.relative('C:\\Program Files\\Call of Duty', 'C:\\');              // -> '../..'                       // returns the relative path between the passed paths assuming the 1st argument as current directory   
    path.win32.relative('C:\\Program Files', 'C:\\Program Files\\Call of Duty\\mode');  // -> 'Call of Duty\\mode'
    path.win32relative('', 'C:\\');                                              // -> '..'                          // the ''(zero-length string) is assumed as current directory (we are in 'C:\\windows')   
    path.win32.relative('C:\\Program Files', 'C:\\server');                      // -> '..\server'                   // -! navigates back to the root directory then to the target path 
    path.win32.relative('C:\\Program Files', 'D:\\server');                      // -> 'D:\\server'                  // -! because the root elements are different the target path is returned 
        
    path.posix.relative('/dev/app', '/');                                        // -> '../../'                      // returns the relative path between the passed paths assuming the 1st argument as current directory   
    path.posix.relative('/dev/app', '/dev/app/Mario/level1');                    // -> 'Mario/level1'
    path.posix.relative('', '/');                                                // -> '../'                         // the ''(zero-length string) is assumed as current directory (we are in '/srv')   
    path.posix.relative('/dev/some folder', '/srv');                             // -> '../../srv'                   // -! navigates back to the root directory then to the target path 
    
// isAbsolute() -----------------------------------------------------------------
    path.win32.isAbsolute('C:\\Program Files\\Call of Duty');                    // -> true             
    path.win32.isAbsolute('Program Files\\Call of Duty');                        // -> false
                
    path.posix.isAbsolute('/dev/Call of Duty');                                  // -> true
    path.posix.isAbsolute('dev/Call of Duty');                                   // -> false 
    
// toNamespacedPath() -----------------------------------------------------------
    path.win32.toNamespacedPath('C:\\Program Files\\Call of Duty');              // -> '\\?\C:\Program Files\Call of Duty'
    path.win32.toNamespacedPath('Program Files\\Call of Duty');                  // -> '\\?\C:\windows\Program Files\Call of Duty'     // if a relative path is provided it returns the current directory path (we are in 'C:\\windows') with the passed path at the end   
    
    path.posix.toNamespacedPath('/dev/app/Call of Duty');                        // -> '/dev/app/Call of Duty'                         // on POSIX systems returns the passed argument as it is
