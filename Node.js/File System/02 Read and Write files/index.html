    const fs = require('fs');
    
// createReadStream() -----------------------------------------------------------------
    var readStream = fs.createReadStream('someFile', {emitClose:true});                // create a readable stream of the file (by default it must be an existing file)   // -! file descriptor won't be closed (2nd argument)     
    
    readStream.path;                                                                   // -> 'someFile'             // first (path) argument of the 'createReadStream()' method 
    readStream.pending;                                                                // -> true                   // file descriptor is NOT allocated (data is pending)    
    
    readStream.on('open', function(fd){                                                // event fired when the file descriptor is allocated (opened)   
        fd;                                                                            // -> 3                      // file descriptor (id)   
    });
    
    readStream.on('ready', function(){                                                 // event fires right after the 'open' event  
        readStream.pending;                                                            // -> false                  // file descriptor is allocated (data is not pending)    
    })
    
    readStream.on('close', function(){                                                 // -! event fires twice 1. when stream closes (standard readable stream event) 2. when the file descriptor closes (because it is set)    
        readStream.bytesRead;                                                          // -> 27                     // number of bytes read so far 
        readStream.pending;                                                            // -> true                   // file descriptor is NOT allocated again (data is pending)   
    })
    
    readStream.on('data', () => {});                                                   // reading the stream 
    
// createWriteStream() ----------------------------------------------------------------
    var writeStream = fs.createWriteStream('testFile', {autoClose:false, flags:'r+'}); // creates a writable stream of the file (file descriptor won't be closed when the stream closes and the the targeted file must be an existing file)    
    
    writeStream.path;                                                                  // -> 'testFile'             // first (path) argument of the 'createWriteStream()' method   
    writeStream.pending;                                                               // -> true                   // file descriptor is NOT allocated (data pending)   
    
    writeStream.on('open', function(fd){                                               // event fires when the file descriptor is allocated (opened)   
        fd;                                                                            // -> 4                      // file descriptor (id)  
    });
    
    writeStream.on('ready', function(){                                                // event fires right after the 'open' event   
        writeStream.pending;                                                           // -> false                  // file descriptor is allocated () 
    })
    
    writeStream.on('close', function(){                                                // event fires when the stream closes only (standard writable stream event)   
        writeStream.bytesWritten;                                                      // -> 14                     // number of bytes written so far
        writeStream.pending;                                                           // -> false                  // -! file descriptor stays open after the stream is close (not recommended)   
    })
    
    writeStream.write('new data added');                                               // writting the stream   
    writeStream.destroy();

createReadStream() / createWriteStream() (event) TEST

- here we are dealing with the same file but in 2 channels, so two different file descriptors are allocated for the two different channels

- the close event is a standard stream event but it can be also emitted when the file descriptor closes (must be set!)

    const fs = require('fs');
    
    var wStream = fs.createWriteStream('testFile', {emitClose:false});                             // file descriptor won't emit the close event when closed 
    var rStream = fs.createReadStream('testFile', {emitClose:true});                               // file descriptor emits the 'close' event when closed  
    
// write stream events ----------------------------------------------------------------------------
    wStream.on('open', function(fd){
        console.log( "writable stream 'open' event emitted! (file descriptor = " + fd + ")" );         
    });
    wStream.on('ready', () => console.log( "writable stream 'ready' event emitted!"));
    wStream.on('close', () => {console.log( "writable stream 'close' event emitted!" );});         // the 'close' event emitted once (because only the stream emits it)
    
// read stream events -----------------------------------------------------------------------------
    rStream.on('open', function(fd){
        console.log("readable stream 'open' event emitted! (file descriptor = " + fd + ")" );
    });
    rStream.on('ready', () => console.log("readable stream 'ready' event emitted!"));
    rStream.on('close', function(){                                                                // this 'close' event is emitted twice (because both the stream and the file descriptor emits it)
        console.log("readable stream 'close' event emitted!");
    });
    
// write read stream --- ---------------------------------------------------------------------------
    wStream.write('write something new')
    wStream.end();
    wStream.destroy();
    
    rStream.on('data', function(chunk){
        console.log( chunk.toString() );
        rStream.destroy();
    });
    
    console.log( '------------ end of module ------------' );
