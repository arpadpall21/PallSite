    const fs = require('fs')
    
// watch() / FSWatcher object ---------------------------------------------------------
  // -! this method is recommended over the 'watchFile() / unwatchFile()' methods, especially for watching directories (faster / more reliable / can watch subdirectories more accurately)    
  
    var fsWatcher = fs.watch('testFile', {recursive:true}, function(eventType, filename){   // returns the fsWatcher object (recursive option will watch subdirectories as well)  
        eventType;                                                                     // -> 'change'                   // event name when the change is triggered 
        filename;                                                                      // -> 'testFile'                 // file or directory where the change is happening   
    });
    
  // fsWatcher object ----------------------------------------                         // the returned fsWatcher object watches the file for changes  
    fsWatcher.on('change', function(eventType, filename){                              // event triggered on file|directory change   
        eventType;                                                                     // -> 'change'                   // event name when the change is triggered 
        filename;                                                                      // -> 'testFile'                 // file or directory where the change is happening   
    });
    
    fsWatcher.on('error', function(err){ err; });                                      // event triggered if an error occures while watching the file   
    fsWatcher.on('close', function(){ });                                              // event triggered when the fsWatcher object stops watching the file or directory 
    
    fsWatcher.close();                                                                 // closes the fsWatcher object (file is not watched after this)   
    
// watchFile() / unwatchFile() --------------------------------------------------------
    fs.watchFile('testFile', {interval:3000}, function(current, previous){             // watching the targeted file (check for changes in every 3 seconds, if there are changes the callback functon is called)   
        current;                                                                       // -> Stats {...}                // stats object of the current state (after changes)  
        previous;                                                                      // -> Stats {...}                // stats object of the previous state (before changes)
    });
    
    fs.unwatchFile('testFile');                                                        // stops wathing the targeted file (2nd argument could remove a referenced listener)
    
  // statWatcher object ---------------------------------------------------------------
    var statWatcher_1 = fs.watchFile('./testFile.txt', function(curent, new_){ }       // returns a statWatcher, prevents the process from closing while the file is watched 
    
  // unref() / ref() ------------------------------------------------------------------
        statWatcher_1.unref();                                                         // this statWatcher not prevent the process from closing 
        statWatcher_1.ref();                                                           // (re)attached the statWatcher to the event loop so it prevents the process from closing

    const fs = require('fs');
    
    var watchObj = fs.watch('testDir', {recursive:true}, function(eventType, fileName){ // returns the fsWatcher object (also watches for subdirectory changes)    
        console.log( "callback fired! (event type = " + eventType + ", file name = " + fileName + ")" );
    });
    
// fsWatch object events --------------------------------------------------------------
    watchObj.on('change', function(eventType, fileName){                               // event fires on any change

    });
        
    watchObj.on('error', function(err){                                                // event fires on error  
        if(err) console.log(err);
    });
        
    watchObj.on('close', function(){                                                   // event fires when the fsWatcher object is closed 
        console.log( 'close event fired!' );
    });
    
// stop watching ----------------------------------------------------------------------
    setTimeout(function(){
        watchObj.close();                                                              // closing the fsWatcher object after 10 seconds 
    }, 10000);

The Stats object

- the Stats object holds the current stats of the item (file, directory, ect..) (if it was called by the bigint option the returned values will be expressed in BigInt (big integer) MDN BigInt [link]) 

SYNTAX :    statsObj.isFile()                                                          // returns true if the stats object describes a regular file, otherwise returns false    
            statsObj.isDirectory()                                                     // returns true if the stats object describes a file system directory, otherwise returns false    
            
            statsObj.isSymbolicLink()                                                  // returns true if the stats object describes a symbolik-link, otherwise returns false (works only with the lstat() method!)    
            statsObj.isSocket()                                                        // returns true if the stats object decribes a socket, otherwise returns false 
            
            statsObj.isBlockDevice()                                                   // returns true if the stats object describes a block device, otherwise returns false (block device = some kind of POSIX file [NOT TESTED : 2019-09-30])    
            statsObj.isCharacterDevice()                                               // returns true if the stats object describes a character device, otherwise returns false (character device = some kind of POSIX file [NOT TESTED : 2019-09-30])        
            statsObj.isFIFO()                                                          // returns true if the stats object describes a first-in-first-out pipe, otherwise returns false (FIFO pipe = some kind of POSIX file [NOT TESTED : 2019-09-30])     
            
            statsObj.atime                                                             // returns the last time the file was accessed in a Date object    
            statsObj.mtime                                                             // returns the last time the file was modified in a Date object    
            statsObj.ctime                                                             // returns the last time the file's status was modified in a Date object    
            statsObj.birthtime                                                         // returns the creation time of the file in a Date object 
            
            statsObj.atimeMs                                                           // returns the last time the file was accessed in milliseconds since the POSIX Epoch time    
            statsObj.mtimeMs                                                           // returns the last time the file was modified in milliseconds since the POSIX Epoch time    
            statsObj.ctimeMs                                                           // returns the last time the file's satus was modified in milliseconds since POSIX Epoch time    
            statsObj.birthtimeMs                                                       // returns the creation time of the file in milliseconds since the POSIX Epoch time    
            
            statsObj.atimeNs                                                           // returns the last time the file was accessed in nanoseconds since the POSIX Epoch time (bigint option only)    
            statsObj.mtimeNs                                                           // returns the last time the file was modified in nanoseconds since the POSIX Epoch time (bigint option only)    
            statsObj.ctimeNs                                                           // returns the last time the status of the file was modified in nanoseconds since the POSIX Epoch time (bigint option only)    
            statsObj.birthtimeNs                                                       // returns the creation time of the file in nanoseconds since the POSIX Epoch time (bigint option only)    

            statsObj.size                                                              // returns the file's size in bytes   
            statsObj.blksize                                                           // returns the file system block size (in bytes) for I/O operations  
            statsObj.blocks                                                            // returns the number of file system blocks allocated for this file 

            statsObj.uid                                                               // returns the user id that owns this file (POSIX on windows = 0)
            statsObj.gid                                                               // returns the group id that owns this file (POSIX on windows = 0)    
            
            statsObj.nlink                                                             // returns the number of hard-links for this file (including this file)   
            
            statsObj.dev                                                               // returns the numeric identifier of the device containing this file    
            statsObj.ino                                                               // returns the system specific inode number of the file     
            statsObj.mode                                                              // returns a number which represents a bit-filed describing the file type and mode (no idea what is this)    
            statsObj.rdev                                                              // returns a numeric device id if the file is special (no idea what is this)
