DEMO 
// CSS --------------------------------------------------------------------------------
    #test1 {
        width: 150px;
        height: 50px;
        background-color: lightcoral;
        border: solid 10px lightskyblue;
    }
    
// HMTL -------------------------------------------------------------------------------
    <div id='test1' onclick="show()"> 
        <p> testtesttesttesttesttesttest </p>                                          // element's conent is overflowing both horizontally and vertically   
        <p> test test test test test test test test test test </p>
    </div>
<script>
    let element = document.getElementById('test1');
    
    element.clientHeight;                                                              // -> 50                 // (content + padding)
    element.clientWidth;                                                               // -> 150                // (content + padding)
    
    element.offsetHeight;                                                              // -> 70                 // (content + padding + border)
    element.offsetWidth;                                                               // -> 170                // (content + padding + border)
    
    element.scrollHeight;                                                              // -> 74                 // (content + padding + overflowing content)
    element.scrollWidth;                                                               // -> 181                // (content + padding + overflowing content)
    
    element.clientTop;                                                                 // -> 10                 // top border width    
    element.clientLeft;                                                                // -> 10                 // left border width    
    
    element.getBoundingClientRect();                                                   // -> {    
                                                                                       // ->   x: 10,           // horizontal position relative to the viewport   
                                                                                       // ->   left: 10,        // same as 'x'
                                                                                       // ->   y: 845,          // vertical position relative to the viewport   
                                                                                       // ->   top: 845,        // same as 'y'
                                                                                       // ->   bottom: 965,     // the bottom of the element counted from top the viewport     
                                                                                       // ->   right: 530,      // the right side position of the element counted from left the viewport    
                                                                                       // ->   width: 520,      // element with (content + padding + border)   
                                                                                       // ->   height: 120      // element height (content + padding + border)
                                                                                       // -> }

</script>

Content Scrolling scrollTop / scrollLeft / scroll() / scrollTo() methods

DEMO 
    <div id='test"> 
        // let's suppose whe have some overflowing content here so the div parent element has scrollbars on the UI   
    </div> 
<script> 
    var element = document.getElementById('test2');
    
// scrollLeft / scrollTop -------------------------------------------------------------
    element.scrollLeft;                                                                // -> content horizontally scrolled position within its element in pixels   
    element.scrollTop;                                                                 // -> content vertically scrolled position within its element in pixels   
    
    element.scrollLeft = 100;                                                          // sets the horizontal scrollling position of the content within it element by 100 pixels    
    element.scrollTop = 250;                                                           // sets the vertical scrollling position of the content within it element by 250 pixels  
    
// scroll() / scrollTo() --------------------------------------------------------------
    element.scroll({
        top: 100,                                                                      // scroll the element's content within the element    
        left: 200,
        behavior: 'smooth'
    })
    
    element.scrollTo({                                                                 // this method does the same as above 
        top: 100, 
        left: 200,
        behavior: 'smooth'
    })
</script>
