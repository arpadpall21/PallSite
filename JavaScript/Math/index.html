The abs() / trunc() / sign() / fround() / clz32() methods

DEMO 
// abs() ------------------------------------------------------------------------------
    Math.abs('-1');                                                                    // -> 1
    Math.abs(-2);                                                                      // -> 2
    Math.abs(null);                                                                    // -> 0
    Math.abs('');                                                                      // -> 0
    Math.abs([]);                                                                      // -> 0
    Math.abs([2]);                                                                     // -> 2
    Math.abs([1,2]);                                                                   // -> NaN
    Math.abs({});                                                                      // -> NaN
    Math.abs('string');                                                                // -> NaN
    Math.abs();                                                                        // -> NaN
    
// trunc() / sing() -------------------------------------------------------------------
    Math.trunc('32.32');                                                               // -> 32                     // returns the integer part only  
    Math.trunc([142]);                                                                 // -> 142
    Math.trunc(-415.22);                                                               // -> -415
    
    Math.sign([84552.21]);                                                             // -> 1                      // positive value 
    Math.sign(-441.2);                                                                 // -> -1                     // negative value 
    Math.sign(0);                                                                      // -> 0
    
// fround() ---------------------------------------------------------------------------
    Math.fround(12.12);                                                                // -> 12.119999885559082     // closest 32bit single precision floating point representation    
    Math.fround(1.2);                                                                  // -> 1.2000000476837158
    
// clz32() ----------------------------------------------------------------------------
    Math.clz32(45);                                                                    // -> 26                     // 45 has 26 leading 0bits in 32bit unsigned integer binary representation     
        // 00000000000000000000000000101101                                                // 32bit signed integer binary representation of 45 
    Math.clz32(5548796);                                                               // -> 9
        // 00000000010101001010101011111100                                                // 32bit signed integer binary representation of 5548796
    Math.clz32(-125);                                                                  // -> 0                      // negative values awlays start with 1bit 
        // 11111111111111111111111110000011                                                // 32bit signed integer binary representation of -125
