ISO 8601 date standard

- the ISO 8601 date standard is a Universal Date Standard format, follows a strict standard, exemple: (2017-08-22T12:24:40Z) 

SYNTAX :    YYYY-MM-DDTHH:MM:SS                          
            YYYY-MM-DDTHH:MM:SSZ                                                       // UTC time (Z stands for "zero meridian")
            YYYY-MM-DDTHH:MM:SS+|-HH:MM                                                // UTC time with adjusted time-zone
DEMO 
    1998                                                                               // the year is required only!
    1998-06                    
    1998-06-20                 
    1998-06-20T23:20:06        
    1998-06-02T23:20:06Z                                                               // UTC time 
    1998-06-02T23:20:06+02:00                                                          // time with adjusted time zone (relative to UTC)

The new Date() / Date() constructor

- in JavaScript countes the time in milliseconds from January 1st, 1970, 00:00:00 UTC (called "Zero Time")

- the Date() constructor creates a date object, but the displayed date and time depends on the interface (ex: the Browser displayes the local time)

DEMO 
    new Date(1403256444443);                                                           // -> "Fri Jun 20 2014 12:27:24 GMT+0300 (heure d’été d’Europe de l’Est)"    // milliseconds (zero-time = counterd from January 1st, 1970, 00:00:00 UTC)   
    new Date(2014, 05, 20, 12, 27, 24);                                                // -> "Fri Jun 20 2014 12:27:24 GMT+0300 (heure d’été d’Europe de l’Est)"    // numbers          // -! 5 means June (months are indexed 0 = January)   
    new Date("2014-06-20T12:27:24");                                                   // -> "Fri Jun 20 2014 12:27:24 GMT+0300 (heure d’été d’Europe de l’Est)"    // ISO time 
    new Date("Tue Jun 20 2014 15:27:24 GMT+0300 (heure d’été d’Europe de l’Est)")      // -> "Fri Jun 20 2014 15:27:24 GMT+0300 (heure d’été d’Europe de l’Est)"    // date string 
    
    new Date("2014-06-20T12:27:24Z");                                                  // -> "Fri Jun 20 2014 15:27:24 GMT+0300 (heure d’été d’Europe de l’Est)"    // UTC time specified so converted back local time

Getter Methods getDate() / getDay() / getFullYear() / getHours() / getMilliseconds() / getMinutes() / getMonth() / getSeconds() / getTime() / getUTCDate() / getUTCDay() / getUTCFullYear() / getUTCHours() / getUTCMilliseconds() / getUTCMinutes() / getUTCMonth() / getUTCSeconds() / getTimezoneOffset()

DEMO 
    var myTimeNow = new Date();                                                        // this the locale time right now 
    myTimeNow;                                                                         // -> "Thu Apr 25 2019 18:02:30 GMT+0300 (heure d’été d’Europe de l’Est)"
        
    myTimeNow.getFullYear();                                                           // -> 2019
    myTimeNow.getMonth();                                                              // -> 3              // -! months are indexed (0 = January / 3 = April)   
    myTimeNow.getDate();                                                               // -> 25 
    myTimeNow.getDay();                                                                // -> 4              // -! weekdays are indexed (0 = Sunday / 4 = Thursday)    
    myTimeNow.getHours();                                                              // -> 18
    myTimeNow.getMinutes();                                                            // -> 2
    myTimeNow.getSeconds();                                                            // -> 30
    myTimeNow.getMilliseconds();                                                       // -> 0
    
// UTC --------------------------------------------------------------------------------
    myTimeNow.getUTCFullYear();                                                        // -> 2019
    myTimeNow.getUTCMonth();                                                           // -> 3              // -! months are indexed (0 = January / 3 = April)   
    myTimeNow.getUTCDate();                                                            // -> 25 
    myTimeNow.getUTCDay();                                                             // -> 4              // -! weekdays are indexed (0 = Sunday / 4 = Thursday)    
    myTimeNow.getUTCHours();                                                           // -> 15         
    myTimeNow.getUTCMinutes();                                                         // -> 2 
    myTimeNow.getUTCSeconds();                                                         // -> 30
    myTimeNow.getUTCMilliseconds();                                                    // -> 0
    
// ------------------------------------------------------------------------------------
    myTimeNow.getTime();                                                               // -> 1556204550000  // millisecond value of the date object (UTC time since Zero Time)   
    myTimeNow.getTimezoneOffset();                                                     // -> -180           // time difference between the local and UTC time (relative to local time) (removing 180 minutes from the local time we are getting the UTC time)

The now() / parse() / UTC() / valueOf() / toString() / toUTCString() / toTimeString() / toDateString() / toISOString() / toJSON() / toLocaleDateString() / toLocaleTimeString() / toLocaleString() methods

DEMO 
    var myTime = new Date();  
     
// .now() / .parse() / .UTC() ---------------------------------------------------------
    Date.now();                                                                        // -> 1528390606567      // current UTC time in milliseconds (from 1970 1st January midnight UTC)    
    
    Date.parse("2018-06-07T19:56:46.567");                                             // -> 1528390606567      // ISO 8601 datetime parsed to milliseconds 
    Date.parse('Thu Oct 22 2020 13:50:46 GMT+0300 (Eastern European Summer Time)')     // -> 1603363846000      // datetime string parsed to milliseconds UTC  
    
    Date.UTC(2018, 06, 07, 19, 56, 46, 567);                                           // -> 1530993406567      // passed UTC time converted to milliseconds UTC     
    
// valueOf() -------------------------------------------------------------------------- 
    new Date().valueOf();                                                              // -> 1528390606567      // primitive value of the date object (milliseconds)   
    
// toString() / toUTCString() ---------------------------------------------------------
    myTime.toString();                                                                 // -> "Mon Jun 04 2018 20:01:49 GMT+0300 (GTB Daylight Time)"           
    myTime.toUTCString();                                                              // -> "Mon, 04 Jun 2018 17:01:49 GMT"  
    
// toTimeString() / toDateString() / toISOString() / toJSON() -------------------------
    myTime.toTimeString();                                                             // -> "20:01:49 GMT+0300 (GTB Daylight Time)"
    myTime.toDateString();                                                             // -> "Mon Jun 04 2018"
    myTime.toISOString();                                                              // -> "2018-06-04T17:01:49.000Z"      
    myTime.toJSON();                                                                   // -> "2018-06-04T17:01:49.000Z"
                        
    
// toLocaleString() / toLocaleDateString() / toLocaleTimeString() ---------------------
    new Date().toLocaleString();                                                       // -> "6/4/2018, 8:01:49 PM"     // the returned datetime format is set according to the user's computer setting      
    new Date().toLocaleDateString();                                                   // -> "6/4/2018"                 // the returned datetime format is set according to the user's computer setting      
    new Date().toLocaleTimeString();                                                   // -> "8:01:49 PM"               // the returned datetime format is set according to the user's computer setting
