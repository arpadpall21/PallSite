Description and Demonstration :

- the Reflect object provides methods for interceptable JavaScript operations

- the Reflect object is not a constructor, it has static methods what perform various operations

The get() / set() / construct() / apply() / ownKeys() / has() / getOwnPropertyDescriptor() / defineProperty() / deleteProperty() / getPrototypeOf() / setPrototypeOf() / isExtensible() / preventExtension() methods

DEMO

- Reflect.get() / Reflect.set() 

  const someObj = {p1:21}
  
  Reflect.get(someObj, 'p1');                                                          // -> 21
  Reflect.set(someObj, 'p2', 22);                                                      // -> true  
  
  someObj;                                                                             // -> {p1:21, p2:22}
  
// ------------------------------------------------------------------------------------  
  const someObj = {p1:11}
  const someObj2 = {p1:21}
  
  const proxy = new Proxy(someObj, {
    get(target, prop, receiver){                                                       // the 'receiver' is the object which is passed by Reflect  
      target;                                                                          // -> someObj
      receiver;                                                                        // -> someObj2
      
      return receiver[prop];                                                           // we can return the value of the receiver object's property   
    },
    set(target, prop, val, receiver){
      target;                                                                          // -> someObj
      receiver;                                                                        // -> someObj2
      
      receiver[prop] = val;                                                            // we can set the receiver object's property  
      return true
    }
  })
  
  Reflect.get(proxy, 'p1', someObj2);                                                  // -> 21                      // the get operation is performed on 'someObj2'   
  Reflect.set(proxy, 'p2', 22, someObj2);                                              // -> true                    // the set operation is performed on 'someObj2'   
  
  someObj2;                                                                            // -> {p1:21, p2:22}

// construct() ------------------------------------------------------------------------
  function TestClass(a, b){
    this.p1 = a;
    this.p2 = b;
    
    new.target;                                                                        // -> TestClass3         // references the passed class 
  }
  
  function TestClass3(){}

    return (a + b + c) / 3;
  }

- Reflect.ownKeys() / Reflect.has() / Reflect.getOwnPropertyDescriptor() / Reflect.defineProperty() / Reflect.deleteProperty() 

  const someObj = {p1: 11, p2:21}
  Object.defineProperty(someObj, 'p3', {value:31, enumerable:false})
  
  const someSymbol = Symbol('sym1')
  someObj[someSymbol] = 'symbolVal'
  
  const protObj = {p4: 41}
  Object.setPrototypeOf(someObj, protObj);
  
// ownKeys()---------------------------------------------------------------------------
  Reflect.ownKeys(someObj);                                                            // -> ['p1', 'p2', 'p3', someSymbol]  
  
// has() ------------------------------------------------------------------------------
  Reflect.has(someObj, 'p2');                                                          // -> true 
  Reflect.has(someObj, someSymbol);                                                    // -> true 
  Reflect.has(someObj, 'p4');                                                          // -> true          // because the property is on the prototype chain 
  Reflect.has(someObj, 'p5');                                                          // -> false
  
// getOwnPropertyDescriptor() ---------------------------------------------------------
  Reflect.getOwnPropertyDescriptor(someObj, 'p1');                                     // -> {value:11, writable:true, enumerable:true, configurable:true}
  Reflect.getOwnPropertyDescriptor(someObj, someSymbol);                               // -> {value:'symbolVal', writable:true, enumerable:true, configurable:true}
  Reflect.getOwnPropertyDescriptor(someObj, 'p4');                                     // -> undefined     // inherited property 
  
// defineProperty() -------------------------------------------------------------------
  Reflect.defineProperty(someObj, 'p10', {                                             // -> true  
    value:100, 
    enumerable:true, 
    configurable:true, 
    writable:true
  });  
  
  someObj.p10;                                                                         // -> 100 
  someObj.propertyIsEnumerable('p10');                                                 // -> true 
  
// deleteProperty() -------------------------------------------------------------------
  Reflect.deleteProperty(someObj, 'p1');                                               // -> true 
  Reflect.deleteProperty(someObj, someSymbol);                                         // -> true 
  
  someObj;                                                                             // -> {p2:21, p10:100}

get() / set() TEST 

  const someObj = {
    p1: 21,
    p2: function(x){ return x**2 },
    p3: [1, 2, 3]
  }
  
  console.log( Reflect.get(someObj, 'p1') );                                           // -> 21
  console.log( Reflect.get(someObj, 'p2')(3) );                                        // -> 9
  console.log( Reflect.get(someObj, 'p3') );                                           // -> [1, 2, 3]
  
  console.log( Reflect.set(someObj, 'p1', 11) );                                       // -> true  
  console.log( Reflect.set(someObj, 'p2', function(x){ return x }) );                  // -> true  
  console.log( Reflect.set(someObj, 'p3', {n1:false, n2:10n}) );                       // -> true  
  
  console.log( Reflect.get(someObj, 'p1') );                                           // -> 11
  console.log( Reflect.get(someObj, 'p2')(3) );                                        // -> 3
  console.log( Reflect.get(someObj, 'p3') );                                           // -> {n1:false, n2:10n}

get() / set() last (receiver) argument test TEST 

  const someObj = {p1:11}
  const someObj2 = {p1:21, p2:'someVal'}
  const someObj3 = {p1:31}
  
  const proxy = new Proxy(someObj, {
    get(target, prop, receiver){                                                       // the 'receiver' is the object which is passed by Reflect  
      console.log( target );                                                           // -> someObj
      console.log( receiver );                                                         // -> someObj3
      return receiver[prop];                                                           // we can return the value of the receiver object's property   
    },
    set(target, prop, val, receiver){
      console.log( target );                                                           // -> someObj
      console.log( receiver );                                                         // -> someObj3
      receiver[prop] = val;                                                            // we can set the receiver object's property  
      return true
    }
  })
  
  console.log( Reflect.get(proxy, 'p1', someObj2) );                                   // -> 21                      // the get operation is performed on 'someObj2'    
  console.log( Reflect.set(proxy, 'p1', 51, someObj3));                                // -> true 
  console.log( someObj3 );                                                             // -> {p1:51}                 // receiver's property set  
  
  console.log( Reflect.get(proxy, 'p2', someObj2) );                                   // -> 'someVal' 
  console.log( Reflect.set(proxy, 'p2', 'otherVal', someObj3));                        // -> true
  console.log( someObj3 );                                                             // -> {p1:51, p2:'otherVal'}  // receiver's property set
