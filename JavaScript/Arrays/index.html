- the forEach() method is faster than the map() method because it does not return anything

Description and Demonstration :

- Array items are stored sequentially and can be accessed by their index (memory index = fast data access)

- in JavaScript any data type can be stored in an Array

DEMO 
    ["string", 21, {prop:true}];                                                       // Array literal 
    new Array("string", 21, {prop:true});                                              // does the same as above (but not recommended)    
    
    new Array(21);                                                                     // allocates a 21 item array (array items are allocated as undefined)    
    
    var myArray = [                                                                    // JavaScript Arrays can hold any data types 
        "Pall", 
        21, 
        false, 
        ["Bircsy", 33, false], 
        myFunction, 
        {'1st':"Me", '2nd':"You", '3rd':"he"} 
    ];
    
    function myFunction(a, b) { return a + b; };
    
// calling stored items --------------------------------------------------------------- 
    nyArray[0];                                                                        // -> 21         // array items are indexed 
    nyArray[3][1];                                                                     // -> 'Bircsy'   
    
// add new or replace existing items -------------------------------------------------- 
    myArray[3] = "Array Replaced";                                                     // 4th value overwritten    
    myArray[6] = "new item";                                                           // new value added at index 6  
    
// remove items -----------------------------------------------------------------------
    myArray[2] = undefined;                                                            // 2nd index set to undefined 
    
    delete myArray[5];                                                                 // -> true       // 5th index set to undefined

The isArray() / from() / of() static methods

DEMO 
// isArray() --------------------------------------------------------------------------
    Array.isArray([1, 3]);                                                             // -> true
    Array.isArray({1:3})                                                               // -> false
    Array.isArray("myString")                                                          // -> false
    
// from() -----------------------------------------------------------------------------
    Array.from("Pall");                                                                // -> ["P", "a", "l", "l"]      // string has the "length" property 
    Array.from(new Set([4, false, "myTxt"]));                                          // -> [4, false, "myTxt"]       // "Set" is an iterable object
    Array.from(["egy", "ketto", "harom"], (val, ind) => val = val + " ind = " + ind);  // -> ["egy ind = 0", "ketto ind = 1", "harom ind = 2"]   // callback function used to handle passed values    
    Array.from({prop1:"val", prop2:"val", prop3:[1, 2, 4]})                            // -> []                        // -! JS objects are not iterable objects    
    
// of() -------------------------------------------------------------------------------
    new Array(7);                                                                      // -> [undefined, ...]          // an new Array instance with 7 undefined values created    
    Array.of(7);                                                                       // -> [7]                       // a new Array instance with one value of 7 is created

The pop() / push() / shift() / unshift() methods

DEMO 
    var myArray  = ["Break", "Berlin", "Citadine", "SUV"]; 
    
// pop() ------------------------------------------------------------------------------ 
    myArray.pop();                                                                     // -> "SUV"             // last item is removed and returned 
    myArray;                                                                           // -> ["Break", "Berlin", "Citadine"]
    
// push() -----------------------------------------------------------------------------   
    myArray.push("Convertable", "Sport");                                              // -> 5                 // new items are pushed at the END in the Array and new length is returned    
    myArray;                                                                           // -> ["Break", "Berlin", "Citadine", "SUV", "Convertable", "Sport"]    
    
// shift() ----------------------------------------------------------------------------  
    myArray.shift();                                                                   // -> "Break"           // first item is removed and returned 
    myArray;                                                                           // -> ["Berlin", "Citadine", "SUV"]     
    
// unshift() -------------------------------------------------------------------------- 
    myArray.unshift("Kombi");                                                          // -> 5                 // inserts the passed item at the beginning of the Array and returns the new Array length    
    myArray;                                                                           // -> ["Kombi", "Break", "Berlin", "Citadine", "SUV"]

The splice() / slice() / concat() / flat() methods

DEMO 
    var myArray = ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"];
    var myMultiArray = ['x', ["egy", "ketto", ["un", "deux", ["one", "two"]]]];
    
// splice() ---------------------------------------------------------------------------  
    myArray.splice(1, 0, "Opel", "Ferrary");                                           // -> []                                                 // 2 new items are added, nothing removed 
    myArray;                                                                           // -> ["Audi", "Opel", "Ferrary", "BMW", "Mercedes", "Dacia", "Toyota"]
    
    myArray.splice(2, 2);                                                              // -> ["Ferrary", "BMW"]                                 // 2 items are removed and returned 
    myArray;                                                                           // -> ["Audi", "Opel", "Mercedes", "Dacia", "Toyota"]
    
    myArray.splice(3, 1, "Renault");                                                   // -> ["Dacia"]                                          // 1 item replaced 
    myArray;                                                                           // -> ["Audi", "Opel", "Mercedes", "Renault", "Toyota"]    
        
// slice() ----------------------------------------------------------------------------
    myArray.slice();                                                                   // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]    
    myArray.slice(1, 3);                                                               // -> ["BMW", "Mecedess"]
    myArray;                                                                           // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]     // -! original Array is not modified    
    
// concat() --------------------------------------------------------------------------- 
    myArray.concat("Renault", "Citroen");                                              // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota", "Renault", "Citroen"]  // returns a new array by adding the passed items at the end of it    
    myArray;                                                                           // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]     // -! original Array is not modified    
    
// flat() -----------------------------------------------------------------------------
    myMultiArray.flat();                                                               // -> ["x", "egy", "ketto", ["un", "deux", ["one", "two"]]]   // 1st level depth (default)  
    myMultiArray.flat(2);                                                              // -> ["x", "egy", "ketto", "un", "deux", ["one", "two"]]     // 2nd level depth 
    myMultiArray.flat(3);                                                              // -> ["x", "egy", "ketto", "un", "deux", "one", "two"]       // 3rd level depth
    myMultiArray;                                                                      // -> ['x', ["egy", "ketto", ["un", "deux", ["one", "two"]]]] // -! original Array is not modified

The indexOf() / lastIndexOf() / includes() methods

DEMO 
    var myArray = ["Audi", "BMW", "Mercedes", "Dacia", "BMW", "Toyota", "Renault"];
    
// indexOf() -------------------------------------------------------------------------- 
    myArray.indexOf("BMW");                                                            // -> 1             // searched item is at the 1st index (first found item)   
    myArray.indexOf("BMW", 2);                                                         // -> 4             // search begins from the 2nd index(included)   
    myArray.indexOf("BMW", -2);                                                        // -> -1            // search begins from the 5th index(included) in this situation (item not found)    
    
// lastIndexOf() ----------------------------------------------------------------------
    myArray.lastIndexOf("BMW");                                                        // -> 4             // search starts from the end (first item found index returned)   
    myArray.lastIndexOf("BMW", 4);                                                     // -> 4             // index included   
    
    myArray.lastIndexOf("BMW", 3);                                                     // -> 1             // search starts from the index 3 to the start    
    
// included() -------------------------------------------------------------------------
    myArray.includes("BMW");                                                           // -> true 
    myArray.includes("BMW", 2);                                                        // -> false         // search begins from 2nd index(included)   
    myArray.includes("BMW", 5);                                                        // -> false         // search begins from 5th index(included) in this situation (item not found)

The copyWithin() / fill() methods

DEMO 
    var myArray = ["a", "b", "c", "d", "e", "f", "g"];
    
// copyWithin() -----------------------------------------------------------------------
    myArray.copyWithin(3, 0, 3);                                                       // items from index 0 to 3 replaces items from the index 3
    myArray;                                                                           // -> ["a", "b", "c", "a", "b", "c", "g"]     // original Array is modified 
    
    myArray.copyWithin(6, 0, 3);                                                       // does not scale the array    
    myArray;                                                                           // -> ["a", "b", "c", "a", "b", "c", "a"]     
    
// fill() -----------------------------------------------------------------------------
    myArray.fill("-", 2, 4);                                                           // items replaced between 2 (included) to 4 (excluded) indexes   
    myArray;                                                                           // -> ["a", "b", "-", "-", "e", "f", "g"]     // original Array is modified
    
    myArray.fill("-", 6, 10);                                                         // does not scale the array 
    myArray;                                                                           // -> ["a", "b", "-", "-", "e", "f", "-"]     // original Array is modified

The map() / forEach() methods

- the forEach() method has better performance over the map() method because its callback does not return anything (used for side effects only)

DEMO 
    var myArray = [10, 20, 30];
    
// map() ------------------------------------------------------------------------------
    myArray.map(function(val, index) {                                                 // -> [11, 21, 31]      // new Array returned
        val;                                                                               // -> 10 | 21 | 30      // values 
        index;                                                                             // -> 0  | 1  | 2       // indexes 
        return val +=1;                                                                    // received values are computed and returned 
    });
    
    myArray;                                                                           // -> [10, 20, 30]      // original Array is not modified  
    
// forEach() --------------------------------------------------------------------------
    myArray.forEach(function(val, index) {                                             // -> undefined         // -! callback does not return values therefore this method is faster over the 'map()' method   
        val;                                                                               // -> 10 | 21 | 30      // values 
        index;                                                                             // -> 0  | 1  | 2       // indexes        
        return val +=1;                                                                // -! this statement is ignored in this callback   
    });
    
    myArray;                                                                           // -> [10, 20, 30]      // original Array is not modified

DEMO 
    var myArray = ["egy", "ketto", "harom"];
    
// reduce() ---------------------------------------------------------------------------
    myArray.reduce(function(total, val, index, arr){                                   // -> "egy ketto harom"               // at the end the 'total' value is returned    
        total;                                                                             // -> "egy"   | "egy ketto"

        return total = total + " " + val;                                             
    });
    
// reduceRight() (does the same as above but from end-to-start) -----------------------
    myArray.reduce(function(total, val, index, arr){                                   // -> "harom ketto egy"               // at the end the 'total' value is returned    
        total;                                                                             // -> "harom" | "harom ketto"
        val;                                                                               // -> "ketto" | "egy"  
        
        return total = total + " " + val;                                             
    });

    var iterObj = myArray.keys();                                                      // iterable object returned with the Array's keys (indexes)
