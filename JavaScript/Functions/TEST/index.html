find maximum value function

CODE : 
    <p id="findMax"></p>
<script>
    function findMax() {
        var max = -Infinity;                                // we set the lowest possible number in order to be able to check very low number as well
        for (var i = 0; i < arguments.length; i++) {
            if (arguments[i] > max) {                       // this condition is fires only if the argument is bigger the "max"
                max = arguments[i];
            }
        }
        return max;
    }
    
    document.getElementById('findMax').innerHTML = "find the highest value function = " + findMax(4, 74, 10, -41, 2);
</script>

extracting function from an object

- a function basically is a global object's method (window.method)

- a function in an Object belongs to that object, but you can extract and apply on the global scope easily

CODE : 
     <p id="extract_demo"></p>
<script>
    var potato = {localFunc:function(){return "Executed"}};
    potato.localFunc();                                     // returns "Executed"
    // locaFunc();                                          // this would not work because this method belongs to the "potato" object therefore this object can use it ONLY!
    var extractFunction = potato.localFunc;                 // extracting the function from the local object
    extractFunction();                                      // returns "Executed" (this is basically "window.extractFunction()")
    
    
    document.getElementById('extract_demo').innerHTML =
        " Original Object method = " + potato.localFunc() + "<br>" +
        " Extracted function executed in the global scope = " + extractFunction();
</script>

the bind() method test

CODE : 
    <p id="bind_print"></p>
<script>
"use strict";
// object literal
    var formula = {a:25, b:41, c:11, sum:function(){return this.a + this.b + this.c;}};
    
// global object properties (variables)
    var a = 10;
    var b = 5;
    var c = 20;
    
// bind the object's method to the global scope's properties 
    var addition = formula.sum;                         // extracting and storing the object's function 
    var readyOperation = addition.bind(window);         // bind the function to the object (bind "formula" object's method to the "window" object (global scope)) 
    
// print
    document.getElementById('bind_print').innerHTML = readyOperation();            // execute the prestored formula
</script>

<script>
// create a function (this function is the global object's method (window.average()))
    function average (xx, zz) {                             // the 'xx' an 'zz' can be passed to the function manualy when called
        return (this.a + this.b + this.c + xx + zz) / 5;    // the "this" needs to be set for later use (here it will be the global object "window")
    }

// create global object properties (variables)
    var a = 41;
    var b = 23;
    var c = 78;

// creating objects
    var numberCollection = {a:21, b:83, c:27};
    var numberCollection_2 = {a:87, b:26, c:44};

    var operation_2 = average.bind(numberCollection_2, 84, 14); 
    
// print
