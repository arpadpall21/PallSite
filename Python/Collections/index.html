    dict.get(key)                                                                      // returns the dictionary key value (returns None if the key does not exist)

// list / tuple -----------------------------------------------------------------------    
    list|tuple[idx]                                                                    // returns the item at the specified index 
    list|tuple[-idx]                                                                   // negative indexes are counted from the end (:-1 = second letter / however :-0 is not allowed)
    
    list|tuple[stIdx(inc):endIdx(exc)]                                                 // returns the specified range in a new list|tuple (omitting idx = from the start|to the end)   
    list|tuple[-endIdx(exc):-stIdx(inc)]                                               // negative indexes are counted from the end (:-1 = second last member / however :-0 doesn't work)

// dictionary -------------------------------------------------------------------------
    myDict = {'key1':1, 'key2':2, 'key3':3}
    
    myDict['key3']                                                                     // -> 3
    myDict['key4']                                                                     // -! raises a KeyError because the 'key4' does not exist in the dictionary  
    
    myDict.get('key3')                                                                 // -> 3
    myDict.get('key4')                                                                 // -> None  
    
    
// set --------------------------------------------------------------------------------
    mySet = {1, 2, 3}
    
    for i in mySet:
        i                                                                              // -> 1  |  2  |  3
    
    mySet[0]                                                                           // -! raises a TypeError because sets are not indexed   
    
    
// list / tuple -----------------------------------------------------------------------
    myList = [1, 2, 3, 4, 5]
    myTuple = (1, 2, 3, 4, 5)
    
    myList[1]                                                                           // -> 2           // returns the indexed member    
    myTuple[1]                                                                          // -> 2
    
    myList[-1]                                                                          // -> 5           // element order is counted (not indexed) from the back when using negative indexing 
    myTuple[-2]                                                                         // -> 4
    
    myList[1: 4]                                                                        // -> [2, 3, 4]   // returns the indexed range in a new list|tuple 
    myTuple[1: 4]                                                                       // -> (2, 3, 4)
    
    myList[:2]                                                                          // -> [1, 2]      // range is start to 2nd index excluded   
    myTuple[2:]                                                                         // -> (3, 4, 5)   // range is 2nd index included to the end  
    
    myList[-3:-1]                                                                       // -> [3, 4]      // indexing from the end   
    myTuple[-3:-1]                                                                      // -> (3, 4)
    
    myList[:-2]                                                                         // -> [1, 2, 3]   // from start to end index 2 (excluded)    
    myTuple[-2:]                                                                        // -> (4, 5)      // from end index 2 (indcluded) to the end

Unpacking

- we can assign collection members to variables with unpacking 

    var, ...[, *var

        
 - additional values are collected in var list 

    
] = list|tuple|set|frozenset|dict                                  // members are assigned to variables in the sequence (for dict keys are unpacked)
                                                                                       // as many variables are required as many members are in the list|tuple|set|frozenset|dict (otherwise raises a ValueError)  
                                                                                       // we can use the *val as last variable so remaining members are collected in var list
DEMO 
    a, b, *rest = [1, 2, 3, 4]
    a                                                                                  // -> 1
    b                                                                                  // -> 2
    rest                                                                               // -> [3, 4]
    
    a, b, *rest = (1, 2, 3, 4)
    a                                                                                  // -> 1
    b                                                                                  // -> 2
    rest                                                                               // -> [3, 4]
    
    a, b, *rest = {1, 2, 3, 4}                                                         // -! sets can be unpacked but the order is random   
    a                                                                                  // -> 3
    b                                                                                  // -> 1
    rest                                                                               // -> [2, 4]
    
    a, b, *rest = {'p1':1, 'p2':2, 'p3':3, 'p4':4}                                     // unpacks only keys  
    a                                                                                  // -> 'p1'
    b                                                                                  // -> 'p2'
    rest                                                                               // -> ['p3', 'p4']

list of Comprehension [exp for item in iterable if cond]

- the list of comprehension is a short syntax to create a new list from an iterable based on some kind of condition

DEMO 
    myTuple = (1, 2, 3, 4, 5)
    newList = [x for x in myTuple if x <= 3]                                           // returns in a new list members that are smaller or equal than 3    
    newList                                                                            // -> [1, 2, 3]
    myTuple                                                                            // -> (1, 2, 3, 4, 5)  // does not modify the original iterable  
    
// ------------------------------------------------------------------------------------
    strings = ['The', 'Dog', 'Was', 'Found', 'Back', 'In', 'The', 'House']
    newList = [x.lower() for x in strings if len(x) == 3]                              // format members to lower case and returns them in a new list if their length is 3   
    newList                                                                            // -> ['the', 'dog', 'was', 'the']
    
// ------------------------------------------------------------------------------------
    def isObject(item):                                                                // custom function returns True for object types   
        checkTypes = [list, dict, set, frozenset, tuple]
        for cls in checkTypes:
            if type(item) == cls:
                return True
    
    myItems = [1-2j, 'someString', 1.3, [1, 2], {"dict":1}, 5, {1, 2}]
    primitives = [data for data in myItems if not isObject(data)]
    objects = [data for data in myItems if isObject(data)]
    
    primitives                                                                         // -> [1-2j, 'someString', 1.3, 5]     
    objects                                                                            // -> [[1, 2], {'dict': 1}, {1, 2}]

Dictionaries

- a dictionary is a key/value data type which is ordered (after v3.7), changable and does not allow duplicate keys

- Python dictionaries are associative arrays (keys are hashed under the hood for faster accessing), so dictionary keys can be any hashable data types (primitives and tuples holding only primitives)

DEMO

        'str':'string,',                                                               // primitives ara hashable 
        1:1, 
        1.1:1.1,
        1-1j:1-1j, 
        True: True,
        None: None,
        (1, False): ('some tuple value'),                                              // a tuple can be used as dictionary keys if it holds only hashable types    
        ([1]): 'someVal',                                                              // -! raises a TypeError (tuple cannot hold any unchangable type)
        [1]: 'someVale'                                                                // -! raises a TypeError (list unchangable type)    
    }
    
    dict1['str']
    dict1[(1, False)]
