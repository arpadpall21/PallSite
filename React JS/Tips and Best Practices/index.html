[1] CSS Property Conditional Component Rendering 

    class MyApp extends React.Component{
      render(){
        return( 
          <div>
            <p style={{display: this.props.name ? 'block':'none'}}> Demo Component </p>  // if props.name is false then CSS results in 'dislay:none'    
            <p style={{display: !this.props.name && 'none'}}> Demo Component </p>        // if props.name is false the logical AND returns 'none' which which results in CSS 'display:none'   
          </div>            
        )
      }
    }
    
    ReactDOM.render(<MyApp name="Pall"/>, document.getElementById('testElement'));

[2] Rendering Components based on Array items with array methods 

    import React from 'react';
    import ReactDOM from 'react-dom';
    
    import products from './Products.js';                                              // imported products object is an Array which contains objects 
    
    class MyApp extends React.Component{
      render(){
        var ProductsComponents = products.map(function(val){                           // mapping through Array itesm (object)
          return (
            <div>                                                                      // at every iteration a Component is created with the Array items and returned  
              <p> Product id : {val.id}</p>
              <p> Product Name : {val.name}</p>
              <p> Price : {val.price}</p>
              <p> Descritpion : {val.description}</p>
              <hr />
            </div>
          );
        });
        return( 
          <div>
            {ProductsComponents}                                                       // holds the created components which is rendered as here 
          </div>            
        )
      }
    }
    
    ReactDOM.render(<MyApp name="Pall"/>, document.getElementById('testElement'));
